from js import document, window # Constants SCREEN_WIDTH = 1200 SCREEN_HEIGHT = 800 PADDLE_WIDTH = 25 PADDLE_HEIGHT = 280 BALL_RADIUS = 20 PADDLE_SPEED = 6 BALL_SPEED = 5 # Game State left_paddle = {"x": 50, "y": SCREEN_HEIGHT / 2 - PADDLE_HEIGHT / 2} right_paddle = {"x": SCREEN_WIDTH - 50 - PADDLE_WIDTH, "y": SCREEN_HEIGHT / 2 - PADDLE_HEIGHT / 2} ball = {"x": SCREEN_WIDTH / 2, "y": SCREEN_HEIGHT / 2, "dx": BALL_SPEED, "dy": BALL_SPEED} left_score = 0 right_score = 0 keys = {} game_running = False # HTML Elements canvas = document.getElementById("gameCanvas") ctx = canvas.getContext("2d") start_button = document.getElementById("startGame") resume_button = document.getElementById("resumeGame") restart_button = document.getElementById("restartGame") def reset_ball(): ball["x"] = SCREEN_WIDTH / 2 ball["y"] = SCREEN_HEIGHT / 2 ball["dx"] *= -1 ball["dy"] *= -1 def draw_paddle(paddle): ctx.fillStyle = "white" ctx.fillRect(paddle["x"], paddle["y"], PADDLE_WIDTH, PADDLE_HEIGHT) def draw_ball(): ctx.beginPath() ctx.arc(ball["x"], ball["y"], BALL_RADIUS, 0, 2 * 3.14159) ctx.fillStyle = "white" ctx.fill() ctx.closePath() def draw_scores(): ctx.font = "30px Arial" ctx.fillStyle = "white" ctx.fillText(f"{left_score}", SCREEN_WIDTH / 4, 50) ctx.fillText(f"{right_score}", 3 * SCREEN_WIDTH / 4, 50) def update_ball(): global left_score, right_score ball["x"] += ball["dx"] ball["y"] += ball["dy"] # Ball collision with top and bottom if ball["y"] - BALL_RADIUS < 0 or ball["y"] + BALL_RADIUS > SCREEN_HEIGHT: ball["dy"] *= -1 # Ball collision with paddles if ( ball["x"] - BALL_RADIUS < left_paddle["x"] + PADDLE_WIDTH and left_paddle["y"] < ball["y"] < left_paddle["y"] + PADDLE_HEIGHT ) or ( ball["x"] + BALL_RADIUS > right_paddle["x"] and right_paddle["y"] < ball["y"] < right_paddle["y"] + PADDLE_HEIGHT ): ball["dx"] *= -1 # Ball out of bounds if ball["x"] + BALL_RADIUS < 0: right_score += 1 reset_ball() if ball["x"] - BALL_RADIUS > SCREEN_WIDTH: left_score += 1 reset_ball() def move_paddles(): if keys.get("w") and left_paddle["y"] > 0: left_paddle["y"] -= PADDLE_SPEED if keys.get("s") and left_paddle["y"] + PADDLE_HEIGHT < SCREEN_HEIGHT: left_paddle["y"] += PADDLE_SPEED if keys.get("ArrowUp") and right_paddle["y"] > 0: right_paddle["y"] -= PADDLE_SPEED if keys.get("ArrowDown") and right_paddle["y"] + PADDLE_HEIGHT < SCREEN_HEIGHT: right_paddle["y"] += PADDLE_SPEED def game_loop(): if not game_running: return ctx.clearRect(0, 0, SCREEN_WIDTH, SCREEN_HEIGHT) move_paddles() update_ball() draw_paddle(left_paddle) draw_paddle(right_paddle) draw_ball() draw_scores() window.requestAnimationFrame(game_loop) def start_game(event): global game_running game_running = True start_button.classList.add("hidden") game_loop() def handle_key_down(event): keys[event.key] = True def handle_key_up(event): keys[event.key] = False document.addEventListener("keydown", handle_key_down) document.addEventListener("keyup", handle_key_up) start_button.addEventListener("click", start_game) # Initialize reset_ball()
